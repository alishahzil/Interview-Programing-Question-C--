#include <iostream>
using namespace std;
//insertion sort

int main() 
{
    int arr [] = {2,7,4,1,5,3};
    int size = sizeof(arr)/sizeof(arr[0]);
    
   for(int i = 1; i <=size-1; i++){
     int value = arr[i];
     int hole = i;
     while(hole > 0 && arr[hole-1] > value){
       arr[hole] = arr[hole-1];
       hole--;
     }
     arr[hole] = value;
   }
    
    for (int i =0 ; i < size;i++){
      cout<<arr[i];
      cout<<',';
    }
    cout<<endl;
    
    return 0;
}
