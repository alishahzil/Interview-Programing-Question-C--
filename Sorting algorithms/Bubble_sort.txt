#include <iostream>
using namespace std;
//Bubble sort

int main() 
{
    int arr [] = {2,7,4,1,5,3};
    int size = sizeof(arr)/sizeof(arr[0]);
    
    for(int k = 0 ; k <= size -1; k++){
      for(int i =0;i <= size-2;i++){
        if(arr[i] > arr[i+1]){
          int temp = arr[i+1];
          arr[i+1] = arr[i];
          arr[i] = temp;
        }
      }
    }
    
    for (int i =0 ; i < size;i++){
      cout<<arr[i];
      cout<<',';
    }
    cout<<endl;
    
    return 0;
}
