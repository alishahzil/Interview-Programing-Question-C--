#include <iostream>
using namespace std;

class node{
  public:
   int data;
   node* next;
   node(int val){
     data = val;
     next = NULL;
  }
};
void insertion(node* &head,int val){
  node* n = new node(val);
  if(head == NULL){
    head = n;
    return;
  }
  if(head->data > val){
    n->next = head;
    head = n;
    return;
  }
  node * temp = head;
  
  while(temp->next !=NULL){
    if(temp->next->data < val){
      temp = temp->next;  
    }else{
      break;
    }
  }
  n->next = temp->next;
  temp->next =  n;
  return;
};

void display(node * &head){
  if(head == NULL){
    cout<<"linked list is not created yet"<<endl;
    return;
  }
  
  node * temp = head;
  
  while(temp->next !=NULL){
    cout<<temp->data<<"-->";
    temp = temp->next;
  }
  cout<<temp->data<<endl;
  return;
};

void evenodd(node * &head){
  node *odd = head;
  node *even = head->next;
  node *evenStart = even;
  
  while(odd->next !=NULL && even->next != NULL){
    odd->next = even->next;
    odd = odd->next;
    even->next = odd->next;
    even = even->next;
  }
  
  odd->next = evenStart;
  if(odd->next != NULL){
    even->next;
  }
}
