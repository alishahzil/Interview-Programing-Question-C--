#include <iostream>
using namespace std;

class node{
  public:
   int data;
   node* next;
   node(int val){
     data = val;
     next = NULL;
  }
};
void insertion(node* &head,int val){
  node* n = new node(val);
  if(head == NULL){
    head = n;
    return;
  }
  if(head->data > val){
    n->next = head;
    head = n;
    return;
  }
  node * temp = head;
  
  while(temp->next !=NULL){
    if(temp->next->data < val){
      temp = temp->next;  
    }else{
      break;
    }
  }
  n->next = temp->next;
  temp->next =  n;
  return;
};

void display(node * &head){
  if(head == NULL){
    cout<<"linked list is not created yet"<<endl;
    return;
  }
  
  node * temp = head;
  
  while(temp->next !=NULL){
    cout<<temp->data<<"-->";
    temp = temp->next;
  }
  cout<<temp->data<<endl;
  return;
};

int main() 
{
  node * head = NULL;
  insertion(head,11);
  insertion(head,15);
  insertion(head,20);
  insertion(head,17);
  insertion(head,5);
  insertion(head,27);
  insertion(head,10);
  display(head);
  display(head);
  
  return 0;
}
